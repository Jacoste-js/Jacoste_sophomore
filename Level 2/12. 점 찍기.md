function solution(k, d) {
    let cnt = 0;
    let max = Math.pow(d,2);
    for (let i = 0; i <= d; i += k) {
        let point = Math.floor(Math.sqrt(max - Math.pow(i, 2)));
        cnt += Math.floor((point) / k) + 1;
    }
    return cnt;  
}
