function solution(order) {
    let stack = [];
    let i = 1;
    let cnt = 0;
    order.forEach(e => {
        while (1) {
            if (e === i) {
                cnt++;
                i++;
                break;
            }
            else if (stack[stack.length - 1] === e) {
                stack.pop();
                cnt++;
                break ;
            }
            else
                stack.push(i++);
            if (i > order.length)
                break;
        }
    })
    return cnt;
}
