function solution(order) {
    let stack = [];
    let i = 1;
    let cnt = 0;
    order.forEach(e => {
        while (i <= order.length + 1) {
            if (e === i) i++;
            else if (stack[stack.length - 1] === e) stack.pop();
            else {
                stack.push(i++);
                continue;
            }
            cnt++;
            break;
        }
    })
    return cnt;
}
