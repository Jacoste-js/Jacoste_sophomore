function solution(n) {
    let cnt = 0;
    for (let i = 1; i <= n; i++) {
        let sum = i;
        for (let j = i + 1; sum < n; j++) sum += j;
        if (sum === n) cnt++;
    }
    return cnt;
}

function solution(n) {
    let answer = 1;
    let tmp = 0;
    for (let i = 1; i < n / 2; i++) {
        tmp = 0;
        for (let j = i; j < n ; j++) {
            tmp += j;
            if (tmp > n) break;
            if (tmp === n) {
                answer++;
                break;
            }
        }
    }
    return answer;
}
