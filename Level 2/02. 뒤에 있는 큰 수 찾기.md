// 풀이 시간 이후에 다른 사람의 풀이를 참고한 코드!
function solution(numbers) {
    const stack = [];
    const answer = Array(numbers.length).fill(-1);

    numbers.forEach((el, i) => {
        while (stack.length && numbers[stack.at(-1)] < el)
            answer[stack.pop()] = el;
        stack.push(i);
    })
    return answer;
}
