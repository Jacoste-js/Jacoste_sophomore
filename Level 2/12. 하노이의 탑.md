function solution(n) {
    const ret = [];
    const hanoi = (num, from, to, aux) => {
        if (num === 1) {
            ret.push([from, to]);
            return;
        }
        hanoi(num - 1, from, aux, to);
        hanoi(1, from, to, aux);
        hanoi(num - 1, aux, to, from);
    }
    hanoi(n, 1, 3, 2);
    return ret;
}
