function solution(storey) {
    const num = ("0" + storey).split('').map((el) => Number(el)).reverse();
    const answer = num.reduce((acc, cur, i) => {
        if (cur < 5) return acc + cur;
        if (cur === 5 && num[i + 1] < 5) return acc + cur;
        num[i + 1] += 1;
        return acc + 10 - cur;
    }, 0);
    
    return answer;
}
