function solution(storey) {
    const num = ("0" + storey).split('').map((el) => Number(el)).reverse();
    const answer = num.reduce((acc, cur, i) => {
        if (cur < 5) return acc + cur;
        if (cur === 5 && num[i + 1] < 5) return acc + cur;
        num[i + 1] += 1;
        return acc + 10 - cur;
    }, 0);
    
    return answer;
}

function solution(storey) {
    let cnt = 0;
    let one = 0;
    let num = storey.toString();
    while (num.length) {
        one = Number(num[num.length - 1]);
        cnt += one >= 5 ? 10 - one : one;
        if (num.length === 1) return cnt += one > 5 ? 1 : 0;
        num = num.slice(0, -1);
        if (one >= 5 && !(one === 5 && Number(num[num.length - 1]) < 5)) num = (Number(num) + 1).toString();
    }
}

function solution(storey) {
  let answer = 0;
  const num = storey
    .toString()
    .split('')
    .map((e) => parseInt(e));
  for (let i = num.length - 1; i > -1; --i) {
    if (i === 0) {
      if (num[i] === 10) {
        answer += 1;
      } else if (num[i] < 6) {
        answer += num[i];
      } else {
        answer += 11 - num[i];
      }
    } else if (num[i] < 5) {
      answer += num[i];
    } else if (num[i] === 5) {
      if (num[i - 1] < 5) {
        answer += num[i];
      } else {
        num[i - 1] += 1;
        answer += 5;
      }
    } else if (num[i] === 10) {
      num[i - 1] += 1;
    } else {
      answer += 10 - num[i];
      num[i - 1] += 1;
    }
  }
  return answer;
}
