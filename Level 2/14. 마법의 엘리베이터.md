function solution(storey) {
    let cnt = 0;
    let one = 0;
    let num = storey.toString();
    while (num.length) {
        one = Number(num[num.length - 1]);
        cnt += one >= 5 ? 10 - one : one;
        if (num.length === 1) return cnt += one > 5 ? 1 : 0;
        num = num.slice(0, -1);
        if (one >= 5 && !(one === 5 && Number(num[num.length - 1]) < 5)) num = (Number(num) + 1).toString();
    }
    return cnt;
}
