function solution(id_list, report, k) {
    const countEe = {};
    const countEr = {};
    
    id_list.forEach((er) => countEr[er] = []);
    new Set(report).forEach((el) => {
        const [er, ee] = el.split(" ");
        countEe[ee] = (countEe[ee] || 0) + 1;
        countEr[er].push(ee);
    });
    return id_list.map((er) => countEr[er].filter((el) => countEe[el] >= k).length);
}
