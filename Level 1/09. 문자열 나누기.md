function solution(s) {
    let ret = 0;
    for(let i = 0, a = 0, b = 0, c = s[0]; i + 1 < s.length; i++) {
        if (s[i] === c) a++;
        else b++;
        if (a !== b) continue;
        a = 0, b = 0, c = s[i + 1], ret++;
    }
    return ret + 1;
}

function solution(s) {
    let answer = 0;
    let x = s[0];
    let xCnt = 0;
    let otherCnt = 0;
    [...s].forEach((e)=>{
        if (xCnt === 0) x = e;
        e === x ? xCnt++ : otherCnt++;
        if (xCnt && xCnt === otherCnt){
            xCnt = 0;
            otherCnt = 0;
            answer++;
        }
    });
    if (xCnt) answer++;
    return answer;
}
