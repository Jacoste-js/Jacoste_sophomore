function solution(s) {
    let answer = 0;
    let x = s[0];
    let xCnt = 0;
    let otherCnt = 0;
    [...s].forEach((e)=>{
        if (xCnt === 0) x = e;
        e === x ? xCnt++ : otherCnt++;
        if (xCnt && xCnt === otherCnt){
            xCnt = 0;
            otherCnt = 0;
            answer++;
        }
    });
    if (xCnt) answer++;
    return answer;
}
