function solution(k, score) {
    const answer = [];
    const acc = [];
    
    score.forEach((el, i) => {
        acc.push(el);
        acc.sort((a, b) => b - a);
        if (i <= k - 1) 
            answer.push(acc.at(-1));
        else
            answer.push(acc.at(k - 1));
    })
    return answer;
}
