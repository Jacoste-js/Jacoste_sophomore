function solution(s, skip, index) {
    let c = "abcdefghijklmnopqrstuvwxyz";
    let answer = '';
    for (let i = 0; i < skip.length; i++)
        c = c.replace(skip[i], '');
    for (let i = 0; i < s.length; i++) {
        let idx = c.indexOf(s[i]) + index;
        while (idx >= c.length) idx -= c.length;
        answer += c[idx];
    }
    return answer;
}
