function solution(keymap, targets) {
    let ret = [];
    targets.forEach(e => {
        let tmp, idx, arr = [], cnt = 0;
        for (let i = 0; i < e.length; i++) {
            for (let j = 0; j < keymap.length; j++) {
                idx = keymap[j].indexOf(e[i]);
                if (idx === -1) idx = 100;
                arr.push(idx + 1);
                if (idx === 0) break;
            }
            tmp = Math.min(...arr);
            if (tmp !== 101) cnt += (tmp);
            else {
                cnt = -1;
                break ;
            }
            arr = [];
        }
        ret.push(cnt);
    })
    return ret;
}
