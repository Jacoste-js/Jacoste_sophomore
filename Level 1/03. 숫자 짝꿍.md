    let arrX = [...X].sort((a, b) => b - a);
    let arrY = [...Y].sort((a, b) => b - a);
    let answer = "";
    let j = 0;
    let Ylen = arrY.length;
    for (let i = 0; i < arrX.length; i++) {
        if (j > Ylen - 1) break;
        if (arrX[i] > arrY[j])
            continue;
        else if (arrX[i] === arrY[j]) {
            answer += arrX[i];
            j++;
        }
        else {
            while (arrY[j] > arrX[i] && j < Ylen)
                j++; 
            if (arrX[i] === arrY[j]) { answer += arrX[i]; j++;}
        }
    }
    if (!answer) return "-1";
    if (answer == 0) return "0";
    return answer;
