function solution(wallpaper) {
    let [top, left, bottom, right] = [-1, -1, -1, -1];
    
    wallpaper.forEach((el, i) => {
        if (!el.includes("#")) return;
        if (top == -1) {
            top = i;
            left = el.indexOf("#");
            right = el.lastIndexOf("#") + 1;
            bottom = i + 1;
            return;
        }
        left = Math.min(el.indexOf("#"), left);
        right = Math.max(el.lastIndexOf("#") + 1, right);
        bottom = i + 1;
    })
    return [top, left, bottom, right];
}

function solution(wallpaper) {
    let answer = [-1, -1, -1, -1];
    for (let i = 0; i < wallpaper.length; i++) {
        if (wallpaper[i].includes('#')) {
            answer[0] = i;
            break;
        }
    }
    for (let i = wallpaper.length - 1; i >= 0; i--) {
        if (wallpaper[i].includes('#')) {
            answer[2] = i + 1;
            break;
        }
    }
    for (let j = 0; j < wallpaper[0].length; j++) {
        for (let i = 0; i < wallpaper.length;  i++) {
            if (wallpaper[i][j] === '#') {
                answer[1] = j;
                break;
            }
        }
        if (answer[1] !== -1) break;
    }
    for (let j = wallpaper[0].length - 1; j >= 0; j--) {
        for (let i = wallpaper.length - 1; i >= 0; i--) {
            if (wallpaper[i][j] === '#') {
                answer[3] = j + 1;
                break;
            }
        }
        if (answer[3] !== -1) break;
    }
    return answer;
}
