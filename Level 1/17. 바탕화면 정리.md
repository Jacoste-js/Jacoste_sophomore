//이렇게밖에 못푸는 게 슾ㄹ퍼요  
function solution(wallpaper) {
    let answer = [-1, -1, -1, -1];
    for (let i = 0; i < wallpaper.length; i++) {
        if (wallpaper[i].includes('#')) {
            answer[0] = i;
            break;
        }
    }
    for (let i = wallpaper.length - 1; i >= 0; i--) {
        if (wallpaper[i].includes('#')) {
            answer[2] = i + 1;
            break;
        }
    }
    for (let j = 0; j < wallpaper[0].length; j++) {
        for (let i = 0; i < wallpaper.length;  i++) {
            if (wallpaper[i][j] === '#') {
                answer[1] = j;
                break;
            }
        }
        if (answer[1] !== -1) break;
    }
    for (let j = wallpaper[0].length - 1; j >= 0; j--) {
        for (let i = wallpaper.length - 1; i >= 0; i--) {
            if (wallpaper[i][j] === '#') {
                answer[3] = j + 1;
                break;
            }
        }
        if (answer[3] !== -1) break;
    }
    return answer;
}
