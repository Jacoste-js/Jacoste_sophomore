        if (i * i == n) answer++;
        else if (!(n % i)) answer += 2;

function divisorCal(n) {
	let cnt = 0;

	for (let i = 1; i * i <= n; i++) {
		if (i * i === n) {
			cnt += 1;
		} else if (n % i === 0) {
			cnt += 2;
		}
	}
	return cnt;
}

function solution(number, limit, power) {
	var answer = 0;
	let totalSteel = 0;

	for (let i = 1; i < number + 1; i++) {
		const wp = divisorCal(i);
		(wp > limit) ? totalSteel += power : totalSteel += wp;
	}
	answer = totalSteel;
	return answer;
}
