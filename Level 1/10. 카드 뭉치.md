function solution(cards1, cards2, goal) {
    for (let i = 0; i < goal.length; i++) {
        if (cards1[0] === goal[i]) cards1.shift();

function solution(cards1, cards2, goal) {
    let i = 0, i1 = 0, i2 = 0;
    for (; i < goal.length; i++) {
        if (goal[i] === cards1[i1]) i1++;
        else if (goal[i] === cards2[i2]) i2++;
        else break;
    }
    return i === goal.length ? "Yes" : "No";
}
