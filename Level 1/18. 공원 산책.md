function solution(park, routes) {
    let [h, w] = [0, 0];
    let [hmax, wmax] = [park.length, park[0].length];
    for (let i = 0; i < park.length; i++) {
        let idx = park[i].indexOf('S');
        if (idx !== -1) [h, w] = [i, +idx];
    }
    routes.forEach(e => {
        let [dir, len] = e.split(' ');
        len = +len;
        if (dir === 'E') {
            if (w + len >= wmax) return;
            for (let i = 1; i <= len; i++)
                if (park[h][w + i] === 'X') return;
            w += len;
        } else if (dir === 'W') {
            if (w - len < 0) return;
            for (let i = 1; i <= len; i++)
                if (park[h][w - i] === 'X') return;
            w -= len;
        } else if (dir === 'S') {
            if (h + len >= hmax) return;
            for (let i = 1; i <= len; i++)
                if (park[h + i][w] === 'X') return;
            h += len;
        } else {
            if (h - len < 0) return;
            for (let i = 1; i <= len; i++)
                if (park[h - i][w] === 'X') return;
            h -= len;
        }
    })
    return [h, w];
}
